Creating and modifying basic array types

Python Go 
# Create a new list
python_list = [1, 2, 3]

# Print list
print(python_list)
#[1, 2, 3]

# Add to list
python_list.append(4)
#[1, 2, 3, 4]

# Iterate over list
for i in python_list:
    print(i)



package main

import "fmt"

func main() {
	// Create a new array
	go_array := []int{1, 2, 3}

	// Print array
	fmt.Println(go_array)
	//[1 2 3]

	// Add to array
	go_array = append(go_array, 4)
	//[1 2 3 4]

	for _, v := range go_array {
		fmt.Println(v)
	}
}
